## Usage

import { Tab, Tabs } from 'fumadocs-ui/components/tabs';

<Tabs items={['Auto Form', 'Manual']}>
  <Tab value="Auto Form">
```ts
import * as z from 'zod';
import { zodLanguageInput } from 'vitnode-frontend/helpers/zod';
import { AutoForm } from 'vitnode-frontend/form/auto-form';
import { AutoFormStringLanguageInput } from 'vitnode-frontend/form/fields/text-language-input';
```

```ts
const formSchema = z.object({
  description: zodLanguageInput,
});
```

```tsx
<AutoForm
  formSchema={formSchema}
  fields={[
    {
      id: 'description',
      component: AutoFormStringLanguageInput,
    },
  ]}
/>
```

  </Tab>

  <Tab value="Manual">
```ts
import { StringLanguageInput } from 'vitnode-frontend/components/ui/text-language-input'; 
```

```tsx
<StringLanguageInput onChange={onChange} value={value} />
```

  </Tab>
</Tabs>
