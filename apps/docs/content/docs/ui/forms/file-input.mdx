## Usage

import { Tab, Tabs } from 'fumadocs-ui/components/tabs';

<Tabs items={['Auto Form', 'Manual']}>
  <Tab value="Auto Form">
```ts
import * as z from 'zod';
import { AutoForm } from 'vitnode-frontend/form/auto-form';
import { AutoFormFileInput } from 'vitnode-frontend/form/fields/file';
import { zodFile } from 'vitnode-frontend/helpers/zod';
```

```ts
const formSchema = z.object({
  logo: zodFile, // [!code highlight]
});
```

```tsx
<AutoForm
  formSchema={formSchema}
  fields={{
    id: 'logo',
    component: AutoFormFileInput,
    componentProps: {
      acceptExtensions: ['png', 'jpg', 'gif'],
      maxFileSizeInMb: 2,
      accept: 'image/png, image/gif, image/jpeg',
      showInfo: true,
    } as AutoFormFileInputProps,
  }}
/>
```

  </Tab>

  <Tab value="Manual">
```ts
import { FileInput } from 'vitnode-frontend/components/ui/file-input';
```

```tsx
<FileInput onChange={onChange} value={value} />
```

  </Tab>
</Tabs>

### Multiple Files

<Tabs items={['Auto Form', 'Manual']}>
  <Tab value="Auto Form">
```ts
import * as z from 'zod';
import { AutoForm } from 'vitnode-frontend/form/auto-form';
import { zodFile } from 'vitnode-frontend/helpers/zod';
```

```ts
const formSchema = z.object({
  logo: z.array(zodFile), // [!code highlight]
});
```

```tsx
<AutoForm
  formSchema={formSchema}
  fields={[
    {
      id: 'logo',
      component: AutoFormFileInput,
      componentProps: {
        acceptExtensions: ['png', 'jpg', 'gif'],
        maxFileSizeInMb: 2,
        accept: 'image/png, image/gif, image/jpeg',
        showInfo: true,
        multiple: true, // [!code highlight]
      } as AutoFormFileInputProps,
    },
  ]}
/>
```

  </Tab>

  <Tab value="Manual">
```ts
import { FileInput } from 'vitnode-frontend/components/ui/file-input';
```

```tsx
<FileInput onChange={onChange} value={value} multiple />
```

  </Tab>
</Tabs>
