## Usage

import { Tab, Tabs } from 'fumadocs-ui/components/tabs';

<Tabs items={['Auto Form', 'Manual']}>
  <Tab value="Auto Form">
```ts
import * as z from 'zod';
import { AutoForm } from 'vitnode-frontend/form/auto-form';
import { AutoFormRadioGroup, AutoFormRadioGroupProps } from 'vitnode-frontend/form/fields/radio-group';
````

```ts
const formSchema = z.object({
  type: z.enum(['option-one', 'option-two']),
});

// or

const formSchema = z.object({
  type: z.nativeEnum(RadioGroupExampleEnum),
});
```

```tsx
<AutoForm
  formSchema={formSchema}
  fields={[
    {
      id: 'type',
      component: AutoFormRadioGroup,
      componentProps: {
        labels: {
          ['option-one']: {
            title: 'Option One',
            description: 'Option One Description',
          },
          ['option-two']: {
            title: 'Option Two',
            description: 'Option Two Description',
          },
        },
      } as AutoFormRadioGroupProps,
    },
  ]}
/>
```

  </Tab>

  <Tab value="Manual">
```ts
import { Label } from 'vitnode-frontend/components/ui/label';
import {
  RadioGroup,
  RadioGroupItem,
} from 'vitnode-frontend/components/ui/radio-group';
```

```tsx
<RadioGroup defaultValue="option-one">
  <div className="flex items-center space-x-2">
    <RadioGroupItem value="option-one" id="option-one" />
    <Label htmlFor="option-one">Option One</Label>
  </div>
  <div className="flex items-center space-x-2">
    <RadioGroupItem value="option-two" id="option-two" />
    <Label htmlFor="option-two">Option Two</Label>
  </div>
</RadioGroup>
```

  </Tab>
</Tabs>
