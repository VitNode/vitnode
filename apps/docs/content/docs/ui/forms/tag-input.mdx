## Usage

import { Tab, Tabs } from 'fumadocs-ui/components/tabs';

<Tabs items={['Auto Form', 'Manual']}>
  <Tab value="Auto Form">
```ts
import * as z from 'zod';
import { AutoForm } from 'vitnode-frontend/form/auto-form';
import { AutoFormTagInput } from 'vitnode-frontend/form/fields/tags';
import { zodTag } from 'vitnode-frontend/helpers/zod';
```

```ts
const formSchema = z.object({
  keywords: zodTag,
});
```

```tsx
<AutoForm
  formSchema={formSchema}
  fields={[
    {
      id: 'keywords',
      component: AutoFormTagInput,
    },
  ]}
/>
```

  </Tab>

  <Tab value="Manual">
```ts
import { TagInput } from 'vitnode-frontend/components/ui/tag-input';
```

```tsx
<TagInput onChange={onChange} value={value} />
```

  </Tab>
</Tabs>

### Multiple

```tsx
<AutoForm
  formSchema={formSchema}
  fields={[
    {
      id: 'keywords',
      component: AutoFormTagInput,
      componentProps: {
        multiple: true,
      } as AutoFormTagsProps,
    },
  ]}
/>
```
