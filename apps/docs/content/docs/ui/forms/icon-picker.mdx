## Usage

import { Tab, Tabs } from 'fumadocs-ui/components/tabs';

<Tabs items={['Auto Form', 'Manual']}>
  <Tab value="Auto Form">
```ts
import * as z from 'zod';
import { AutoForm } from 'vitnode-frontend/form/auto-form';
import { AutoFormIconPicker } from 'vitnode-frontend/form/fields/icon';
```

```ts
const formSchema = z.object({
  icon: z.string(),
});
```

```tsx
<AutoForm
  formSchema={formSchema}
  fields={[
    {
      id: 'icon',
      component: AutoFormIconPicker, // [!code highlight]
    },
  ]}
/>
```

  </Tab>

  <Tab value="Manual">
```ts
import { IconPicker } from 'vitnode-frontend/components/icon/picker/icon-picker';
```

```tsx
<IconPicker onChange={onChange} value={value} />
```

  </Tab>
</Tabs>
