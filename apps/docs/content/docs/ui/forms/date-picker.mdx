## Usage

import { Tab, Tabs } from 'fumadocs-ui/components/tabs';

<Tabs items={['Auto Form', 'Manual']}>
  <Tab value="Auto Form">
```ts
import * as z from 'zod';
import { AutoForm } from 'vitnode-frontend/form/auto-form';
import { AutoFormDatePicker } from 'vitnode-frontend/form/fields/date';
```

```ts
const formSchema = z.object({
  birthday: z.date().default(new Date()),
});
```

```tsx
<AutoForm
  formSchema={formSchema}
  fields={[
    {
      id: 'birthday',
      component: AutoFormDatePicker,
    },
  ]}
/>
```

  </Tab>

  <Tab value="Manual">

```tsx
'use client';

import React from 'react';
import { DatePicker } from 'vitnode-frontend/components/ui/date-picker';

export function DatePickerDemo() {
  const [date, setDate] = React.useState<Date | undefined>(new Date());

  return <DatePicker value={date} onChange={setDate} />;
}
```

  </Tab>
</Tabs>

## Examples

### Controlled Month and Year

Enable dropdown for month and year.

```tsx
<AutoForm
  formSchema={formSchema}
  fields={[
    {
      id: 'birthday',
      component: AutoFormDatePicker,
      componentProps: {
        controlledMouthAndYear: {
          fromYear: 1900,
          toYear: new Date().getFullYear(),
        },
      } as AutoFormDatePickerProps,
    },
  ]}
/>
```
