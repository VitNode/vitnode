## Usage

import { Tab, Tabs } from 'fumadocs-ui/components/tabs';

<Tabs items={['Auto Form', 'Manual']}>
  <Tab value="Auto Form">
```ts
import * as z from 'zod';
import { AutoForm } from 'vitnode-frontend/form/auto-form';
import { AutoFormInput } from 'vitnode-frontend/form/fields/input';
```

```ts
const formSchema = z.object({
  username: z.string(),
});
```

```tsx
<AutoForm
  formSchema={formSchema}
  fields={[
    {
      id: 'username',
      component: AutoFormInput, // [!code highlight]
    },
  ]}
/>
```

  </Tab>

  <Tab value="Manual">
```ts
import { Input } from 'vitnode-frontend/components/ui/input';
```

```tsx
<Input />
```

  </Tab>
</Tabs>

### With placeholder

<Tabs items={['Auto Form', 'Manual']}>
  <Tab value="Auto Form">

```tsx
<AutoForm
  formSchema={formSchema}
  fields={[
    {
      id: 'name',
      component: AutoFormInput,
      componentProps: {
        placeholder: 'Type your name', // [!code highlight]
      } as AutoFormInputProps,
    },
  ]}
/>
```

  </Tab>

  <Tab value="Manual">

```tsx
<Input placeholder="Email" />
```

  </Tab>
</Tabs>
