## Usage

```tsx
import {
  AlertDialog,
  AlertDialogAction,
  AlertDialogCancel,
  AlertDialogContent,
  AlertDialogDescription,
  AlertDialogFooter,
  AlertDialogHeader,
  AlertDialogTitle,
  AlertDialogTrigger,
} from 'vitnode-frontend/components/ui/alert-dialog';
```

```tsx
<AlertDialog>
  <AlertDialogTrigger>Open</AlertDialogTrigger>
  <AlertDialogContent>
    <AlertDialogHeader>
      <AlertDialogTitle>Are you absolutely sure?</AlertDialogTitle>
      <AlertDialogDescription>
        This action cannot be undone. This will permanently delete your account
        and remove your data from our servers.
      </AlertDialogDescription>
    </AlertDialogHeader>
    <AlertDialogFooter>
      <AlertDialogCancel>Cancel</AlertDialogCancel>
      <AlertDialogAction>Continue</AlertDialogAction>
    </AlertDialogFooter>
  </AlertDialogContent>
</AlertDialog>
```

### Lazy Load Content

To improve performance and reduce the size of the initial bundle use [React Lazy](https://react.dev/reference/react/lazy) to lazy load the dialog content. Use this if you have a large components inside the dialog.

<Step>
 
#### Trigger & Basic Content

```tsx title="test.tsx"
import {

  AlertDialogContent,
  AlertDialogTrigger,
} from 'vitnode-frontend/components/ui/alert-dialog';
import { Button } from 'vitnode-frontend/components/ui/button';

export const Test = () => {

```

</Step>

<Step>

#### Content

Your dialog content should be in a separate file, for example `content.tsx`.

```tsx title="content.tsx"
import {
  AlertDialogAction,
  AlertDialogCancel,
  AlertDialogContent,
  AlertDialogDescription,
  AlertDialogFooter,
  AlertDialogHeader,
  AlertDialogTitle,
} from 'vitnode-frontend/components/ui/alert-dialog';

export const ContentTest = () => {
  return (
    <>
      <AlertDialogHeader>
        <AlertDialogTitle>This is title</AlertDialogTitle>
        <AlertDialogDescription>This is description</AlertDialogDescription>
      </AlertDialogHeader>
      <AlertDialogFooter>
        <AlertDialogCancel>Cancel</AlertDialogCancel>
        <AlertDialogAction>Continue</AlertDialogAction>
      </AlertDialogFooter>
    </>
  );
};
```

</Step>

#### Lazy Load Content

```tsx title="test.tsx"
import React from 'react';

import {
  AlertDialog,
  AlertDialogContent,
  AlertDialogTrigger,
} from 'vitnode-frontend/components/ui/alert-dialog';

const Content = React.lazy(async () =>
  import('./content').then(module => ({
    default: module.ContentTest,
  })),
);

export const Test = () => {
  return (
    <AlertDialog>
      <AlertDialogTrigger asChild>
        <Button variant="ghost" size="icon" ariaLabel="Edit">
          <Pencil />
        </Button>
      </AlertDialogTrigger>

      <AlertDialogContent>
        <React.Suspense fallback={<Loader />}>
          <Content />
        </React.Suspense>
      </AlertDialogContent>
    </AlertDialog>
  );
};
```

## Hook State

We're created context to control dialog state `open` and `setOpen`. You can use it as hook inside dialog component.

```tsx
import { useAlertDialog } from 'vitnode-frontend/components/ui/alert-dialog';
```

```tsx
const { open, setOpen } = useAlertDialog();
```
