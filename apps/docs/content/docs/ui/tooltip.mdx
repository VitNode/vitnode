---
title: Tooltip
description: A popup that displays information related to an element when the element receives keyboard focus or the mouse hovers over it.
---

  This component is part of [Shadcn
  UI](https://ui.shadcn.com/docs/components/tooltip).
</Callout>

## Usage

import { Tab, Tabs } from 'fumadocs-ui/components/tabs';

<Tabs items={['Wrapper', 'Manual']}>
  <Tab value="Wrapper">
```tsx
import { TooltipWrapper } from 'vitnode-frontend/components/ui/tooltip';
```

```tsx
<TooltipWrapper content={<p>Add to library</p>}>
  <Button>Hover</Button>
</TooltipWrapper>
```

  </Tab>

  <Tab value="Manual">
```tsx
import {
  Tooltip,
  TooltipContent,
  TooltipProvider,
  TooltipTrigger,
} from 'vitnode-frontend/components/ui/tooltip';
```

```tsx
<TooltipProvider>
  <Tooltip>
    <TooltipTrigger>Hover</TooltipTrigger>
    <TooltipContent>
      <p>Add to library</p>
    </TooltipContent>
  </Tooltip>
</TooltipProvider>
```

  </Tab>
</Tabs>
