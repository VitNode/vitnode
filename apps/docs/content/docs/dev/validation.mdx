Below you can find an example of how to validate incoming data.

## Body

```ts title="show/show.args.ts"
import { IsString, IsOptional, IsInt } from 'class-validator';

export class ShowNavExampleBody {
  @IsString()
  @IsOptional()
  name?: string;

  @IsInt()
  id: number;
}
```

```ts title="example.controller.ts"
@Controller('example/nav')
export class NavExampleController {
  constructor(private readonly showService: ShowCoreNavService) {}

  @Get()
  // [!code word:ShowNavExampleBody]
  async findAll(@Body() args: ShowNavExampleBody): Promise<ShowCoreNavObj> {
    return this.showService.show(args);
  }
}
```
